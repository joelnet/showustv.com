# Only the landing page is meant for crawlers (issue #208). Everything else
# is a signed-in app surface or a per-user page. "Allow: /$" wins over
# "Disallow: /" for the exact root URL because allow/disallow precedence goes
# to the most specific (longest) matching rule (RFC 9309), and /assets, /icons,
# /og.png and the manifest stay fetchable so renderers can actually draw the
# landing page and social cards. "Allow: /?" (a literal ?) keeps landing-page
# links with query strings (/?utm_source=...) crawlable too — matching runs
# against path + query, so "/$" alone wouldn't cover them.
User-agent: *
Allow: /$
Allow: /?
Allow: /assets/
Allow: /icons/
Allow: /og.png
Allow: /manifest.webmanifest
Disallow: /

# Social link unfurlers (issue #211): several of these honor robots.txt when
# fetching a shared URL for its card (Twitterbot, facebookexternalhit,
# Slackbot notably), so without their own group the per-page previews would
# never render there. Same rules as the "*" group above plus the shareable
# pages the Worker rewrites: title pages (/show, /movie, /episode — issue
# #211) and everything under /u/ — public profiles (issue #219) and shared
# lists (issue #335). Allowing all of /u/ is safe: the Worker serves the
# generic landing card for a private/unknown profile or an unshared/unknown
# list, so an unfurler can't confirm or leak one. Everything else (API and the
# signed-in app surfaces) stays off-limits to them too (issue #208).
User-agent: Twitterbot
User-agent: facebookexternalhit
User-agent: Facebot
User-agent: Slackbot
User-agent: LinkedInBot
User-agent: Discordbot
User-agent: TelegramBot
User-agent: WhatsApp
Allow: /$
Allow: /?
Allow: /show/
Allow: /movie/
Allow: /episode/
Allow: /u/
Allow: /assets/
Allow: /icons/
Allow: /og.png
Allow: /manifest.webmanifest
Disallow: /
